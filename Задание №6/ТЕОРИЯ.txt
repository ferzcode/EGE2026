from turtle import *

# С ГРАДУСАМИ
left() | lt()
right() | rt()

# С ПИКСЕЛЯМИ (ПРИВЯЗЫВАЕМ К ПЕРЕМЕННОЙ K)
forward() | fd()
backward() | bk()

# ВСПОМОГАТЕЛЬНЫЕ
tracer(0) - ОТКЛЮЧАЕТ АНИМАЦИЮ ОТРИСОВКИ
update() - ОБНОВЛЕНИЕ ХОЛСТА
done() - ЗАКРЫТИЕ ПРОГРАММЫ НА КРЕСТИК

screensize(7000, 7000) - РАЗМЕР ВАШЕГО ОКНА

# ОТРИСОВКА
penup() | up() - ПОДНЯТЬ ХВОСТ
pendown() | down() - ОПУСТИТЬ ХВОСТ

dot(3, "red") - ПОСТАВИТЬ ТОЧКУ
goto(x, y) - ПЕРЕМЕЩЕНИЕ В ТОЧКУ С КООРД (X, Y)
